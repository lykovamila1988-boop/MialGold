📡 КАК РАБОТАЕТ ЛОГИРОВАНИЕ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

В pipeline.py импортируешь:
   from chain_dashboard import log_chain_start, log_chain_step, log_chain_end

Перед цепочкой:
   log_chain_start(
     chain_id="my_chain_001",
     from_agent="n8n",
     agents=["olya", "marina", "victoria"],
     description="Еженедельный контент-план"
   )

После каждого агента:
   log_chain_step(
     chain_id="my_chain_001",
     agent="marina",
     step_num=2,
     status="done",
     elapsed_ms=130000
   )

После цепочки:
   log_chain_end(
     chain_id="my_chain_001",
     status="ok",
     total_ms=260000
   )

Эти события пишутся в:
   E:\MILA GOLD\mila-office\memory\events.jsonl (JSONL)

Дашборд читает этот файл и показывает данные.


📈 СТРУКТУРА ДАННЫХ В ПАМЯТИ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

events.jsonl — append-only журнал (JSON Lines)

Типы событий:
  • chain:start  — цепочка начала выполняться
  • chain:step   — шаг завершился (агент отработал)
  • chain:end    — цепочка полностью завершилась

Пример:
  {"ts":"2026-06-08T10:30:45Z","kind":"chain:start","payload":{"chain_id":"content_week_...","from_agent":"n8n","agents":["olya","marina","victoria"]}}
  {"ts":"2026-06-08T10:31:10Z","kind":"chain:step","payload":{"chain_id":"content_week_...","agent":"olya","step_num":1,"status":"done","elapsed_ms":85000}}
  {"ts":"2026-06-08T10:33:20Z","kind":"chain:step","payload":{"chain_id":"content_week_...","agent":"marina","step_num":2,"status":"done","elapsed_ms":130000}}
  {"ts":"2026-06-08T10:45:30Z","kind":"chain:end","payload":{"chain_id":"content_week_...","status":"ok","total_ms":900000}}

   http://127.0.0.1:5000/chains
   │
   ├─→ Загружает HTML дашборда
   │
   └─→ JavaScript начинает polling каждые 5 сек
      ├─ fetch /chains/api/active
      ├─ fetch /chains/api/timeline
      ├─ fetch /chains/api/metrics
      └─ fetch /chains/api/history

Flask (chain_dashboard.py):
   /chains/api/* endpoints
   │
   └─→ Читают memory.EVENTS (JSONL)
      ├─ _read_chain_events()
      ├─ _build_active_chains()
      ├─ _build_agent_timeline()
      ├─ _build_chain_details()
      └─ _build_performance_metrics()

Результат: дашборд обновляется в браузере каждые 5 сек


🎁 БОНУСНЫЕ ПРИМЕРЫ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Пример 1: Получить метрики через Python
   import requests
   resp = requests.get("http://127.0.0.1:5000/chains/api/metrics")
   metrics = resp.json()["overall"]
   print(f"Success rate: {metrics['success_rate']:.1f}%")

Пример 2: Отправить алерт, если цепочка упала
   resp = requests.get("http://127.0.0.1:5000/chains/api/history?status=failed")
   failed = resp.json()["chains"]
   for chain in failed:
       send_telegram_alert(f"Цепочка {chain['chain_id']} упала")

Пример 3: Запросить детали конкретной цепочки
   chain_id = "content_week_20260608_103045"
   resp = requests.get(f"http://127.0.0.1:5000/chains/api/details/{chain_id}")
   steps = resp.json()["chain"]["steps"]
   for step in steps:
       print(f"{step['agent']}: {step['elapsed_human']}")


⚠️ ВАЖНОЕ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Дашборд работает только если есть события в memory.EVENTS
   → Добавьте логирование в pipeline.py (см. CHAIN_DASHBOARD_EXAMPLE.py)

2. Дашборд читает весь файл events.jsonl в памяти
   → На production (> 10000 событий) нужна ротация логов

3. API маршруты всегда доступны, даже без логирования
   → Они просто будут пусты

4. Дашборд не сохраняет историю браузера
   → При перезагрузке страницы — новая сессия
   → Данные хранятся в events.jsonl, не в памяти браузера
